通过原型 prototype 给对象添加方法
----------------------------------------
1. 给数组 添加删除某个元素的方法

(1) 只删除第一个遇到的元素，不处理后面的

var arr=[1,2,3,4,3]
arr.remove(3)

Array.prototype.remove = function(s) {
	for (var i = 0; i < this.length; i++) {
		if (s == this[i]){
			return this.splice(i, 1);
		}
	}
}

arr.remove(3) //[3]
arr //[1, 2, 4, 3]
